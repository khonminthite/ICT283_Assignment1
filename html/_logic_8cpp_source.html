13void Logic::LoadDataFileCSV(ifstream &inputFile, Vector<Weather> &WindLog)
14{
15 // Keeps track of the number of elements split from a given string
16 unsigned int recordCount = 0, rowFieldCount = 0, datetimeCount = 0, dateCount = 0, timeCount = 0;
17 // A temp storage for split string
18 string line, rowField, datetimeField, dateField, timeField;
19 // A temp storage for split string
20 Vector<string> columns, splitDateTime, splitDate, splitTime;
21 // Temp storage for Date object
22 Date tempDate;
23 // Temp storage for Time object
24 Time tempTime;
25 // Initializes ValidationHelper
26 Validator Validate;
27 // Temp storage for Weather object
28 Weather tempLog;
29 // input string stream for reading and splitting of elements
30 istringstream row, datetime, date, time;
31
32 // Get and ignore CSV header
33 getline(inputFile, line);
34
35 string columnNames[] = {"WAST", "S", "SR", "T"};
36 int columnIndices[4] = {-1, -1, -1, -1};
37
38 istringstream headerStream(line);
39 string columnName;
40 int columnIndex = 0;
41
42 while (getline(headerStream, columnName, ','))
43 {
44 for (int i = 0; i < 4; i++)
45 {
46 if (columnName == columnNames[i])
47 {
48 columnIndices[i] = columnIndex;
49 break;
50 }
51 }
52 columnIndex++;
53 }
54
55 while (getline(inputFile, line))
56 {
57 if (line != "")
58 {
59 row.str(line);
60 while (getline(row, rowField, ','))
61 {
62 columns.Insert(rowFieldCount, rowField);
63 rowFieldCount++;
64 }
65 row.clear();
66
67 // Skip record if "WAST" column is empty
68 if (columnIndices[0] != -1 && columns[columnIndices[0]] == "")
69 {
70 cout << "Warning: Empty 'WAST' field detected at row " << recordCount + 1 << endl;
71 continue; // Skip to the next record
72 }
73
74 // Assuming "WAST" column is always present, but its fields can be empty
75 datetime.str(columns[columnIndices[0]]);
76 while (getline(datetime, datetimeField, ' '))
77 {
78 splitDateTime.Insert(datetimeCount, datetimeField);
79 datetimeCount++;
80 }
81 datetime.clear();
82
83 // Handle empty date/time fields
84 if (splitDateTime.Size() > 0)
85 {
86 date.str(splitDateTime[0]);
87 while (getline(date, dateField, '/'))
88 {
89 splitDate.Insert(dateCount, RemoveLeadingZeros(dateField));
90 dateCount++;
91 }
92 date.clear();
93
94 if (splitDate.Size() == 3 && Validate.IsValidDate(stoi(splitDate[0]), stoi(splitDate[1]), stoi(splitDate[2])))
95 {
96 tempDate.SetDay(stoi(splitDate[0]));
97 tempDate.SetMonth(stoi(splitDate[1]));
98 tempDate.SetYear(stoi(splitDate[2]));
99 }
100 else
101 {
102 cout << "Warning: Invalid date(WAST) detected at row " << recordCount + 1 << endl;
103 }
104
105 if (splitDateTime.Size() > 1)
106 {
107 time.str(splitDateTime[1]);
108 while (getline(time, timeField, ':'))
109 {
110 splitTime.Insert(timeCount, RemoveLeadingZeros(timeField));
111 timeCount++;
112 }
113 time.clear();
114
115 if (splitTime.Size() == 2 && Validate.IsValid24HourTime(stoi(splitTime[0]), stoi(splitTime[1])))
116 {
117 tempTime.SetHour(stoi(splitTime[0]));
118 tempTime.SetMinute(stoi(splitTime[1]));
119 }
120 else
121 {
122 cout << "Warning: Invalid time(WAST) detected at row: " << recordCount + 1 << endl;
123 }
124 }
125 }
126
127 // Validate and set other fields only if they are not empty
128 for (int i = 1; i < 4; i++) // Start from 1 to skip "WAST"
129 {
130 if (columnIndices[i] != -1 && columns[columnIndices[i]] != "" && Validate.IsNumber(columns[columnIndices[i]]))
131 {
132 switch (i)
133 {
134 case 1: tempLog.SetWindSpeed(stof(columns[columnIndices[i]])); break;
135 case 2: tempLog.SetSolarRadiation(stof(columns[columnIndices[i]])); break;
136 case 3: tempLog.SetAirTemp(stof(columns[columnIndices[i]])); break;
137 }
138 }
139 else if (columns[columnIndices[i]] != "") // Only print warning if the field is not empty
140 {
141 cout << "Warning: Invalid data detected for " << columnNames[i] << " at row: " << recordCount + 1 << endl;
142 }
143 }
144
145 // Insert tempLog into Vector WindLog only if all necessary fields are valid
146 if (datetimeCount > 0) // Assuming at least date is present
147 {
148 tempLog.SetDate(tempDate);
149 tempLog.SetTime(tempTime);
150 WindLog.Insert(recordCount, tempLog);
151 }
152
153 // Clear the temp storage vectors
154 columns.Clear();
155 splitDateTime.Clear();
156 splitDate.Clear();
157 splitTime.Clear();
158
159 // Reset count
160 rowFieldCount = 0;
161 datetimeCount = 0;
162 dateCount = 0;
163 timeCount = 0;
164
165 recordCount++;
166 }
167 }
168}

267float Logic::StandardDeviationWindSpeedMonth(const Vector<Weather>& array, const int month, const int year)
268{
269 Vector<float> windSpeeds;
270 int count = 0;
271
272 for (unsigned int i = 0; i < array.Size(); i++)
273 {
274 if (array[i].GetDate().GetMonth() == month && array[i].GetDate().GetYear() == year)
275 {
276 count++;
277 windSpeeds.Insert(count - 1, array[i].GetWindSpeed());
278 }
279 }
280
281 if (count == 0)
282 {
283 return -1; // No data for the specified month and year
284 }
285
286 float sum = 0.0;
287 for (unsigned int i = 0; i < windSpeeds.Size(); i++)
288 {
289 sum += windSpeeds[i];
290 }
291 float mean = sum / count;
292
293 float sq_sum = 0.0;
294 for (unsigned int i = 0; i < windSpeeds.Size(); i++)
295 {
296 float diff = windSpeeds[i] - mean;
297 sq_sum += diff * diff;
298 }
299 float variance = sq_sum / (count - 1);
300
301 float stdev = sqrt(variance);
302
303 return round(stdev * 100) / 100;
304}

306float Logic::StandardDeviationAirTemperatureMonth(const Vector<Weather>& array, const int month, const int year)
307{
308 Vector<float> airTemperatures;
309 int count = 0;
310
311 // Loop through the Vector to find all entries for the specified month and year
312 for (unsigned int i = 0; i < array.Size(); i++)
313 {
314 if (array[i].GetDate().GetMonth() == month && array[i].GetDate().GetYear() == year)
315 {
316 count++;
317 airTemperatures.Insert(count - 1, array[i].GetAirTemperature()); // Assuming GetAirTemp() returns the air temperature
318 }
319 }
320
321 // If no data for the specified month and year, return -1
322 if (count == 0)
323 {
324 return -1; // No data for the specified month and year
325 }
326
327 float sum = 0.0;
328 // Calculate the sum of all air temperatures
329 for (unsigned int i = 0; i < airTemperatures.Size(); i++)
330 {
331 sum += airTemperatures[i];
332 }
333 float mean = sum / count; // Calculate the mean air temperature
334
335 float sq_sum = 0.0;
336 // Calculate the sum of the squared differences from the mean
337 for (unsigned int i = 0; i < airTemperatures.Size(); i++)
338 {
339 float diff = airTemperatures[i] - mean;
340 sq_sum += diff * diff;
341 }
342 float variance = sq_sum / (count - 1); // Calculate the variance
343
344 // Calculate the standard deviation from the variance
345 float stdev = sqrt(variance);
346
347 return round(stdev * 100) /100;
